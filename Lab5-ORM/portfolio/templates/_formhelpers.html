{% else %} {% set validateStyle = ' is-valid' if reqMethod == 'POST' else ' ' %} {{field(placeholder = placeHolder, class = "form-control"+validateStyle)}} {% if reqMethod == 'POST' %}

{% endif %} {% endif %}
{% endmacro %}
